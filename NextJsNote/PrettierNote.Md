```js
   baseDirectory: __dirname,
});

   ...compat.extends('next/core-web-vitals'),
   {
      ignores: ['node_modules/**', '.next/**', 'out/**', 'build/**', 'next-env.d.ts'],
   },
   {
      plugins: {
         prettier,
      },
      rules: {
         ...prettierConfig.rules,
         'prettier/prettier': 'warn', //? If Format Error then show warning
      },
   },
];

```
